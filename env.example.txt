# ==========================================
# 铸剑乾坤 - 部署环境变量配置
# ==========================================
# 使用方法：将此文件复制为 .env 文件，然后修改相应的配置值

# ==========================================
# 前端配置
# ==========================================
# 前端访问端口（默认 3000）
FRONTEND_PORT=3000

# 前端 API 地址（生产环境通常使用相对路径或完整域名）
# 开发环境: http://localhost:3001/api
# 生产环境: http://your-domain.com/api 或 /api（使用 nginx 代理）
VITE_API_BASE_URL=http://localhost:3001/api

# Socket.io 连接地址
# 开发环境: http://localhost:3001
# 生产环境: http://your-domain.com 或使用当前域名
VITE_SOCKET_URL=http://localhost:3001

# ==========================================
# 后端配置
# ==========================================
# 后端服务端口（默认 3001）
BACKEND_PORT=3001

# 数据库配置（SQLite）
DATABASE_URL=file:./prisma/swordforge.db

# JWT 密钥（生产环境请务必修改为强密码！）
JWT_SECRET=swordforge_super_secret_change_me

# JWT 过期时间
JWT_EXPIRES_IN=2h
JWT_REFRESH_EXPIRES_IN=7d

# 管理员配置
ADMIN_USERNAME=admin
ADMIN_PASSWORD=123456
ADMIN_NAME=系统管理员

# CORS 允许的源（多个用逗号分隔）
# 生产环境请设置为实际的域名，例如: https://your-domain.com,https://www.your-domain.com
CORS_ORIGIN=http://localhost:3000,http://localhost:5173

# ==========================================
# AI 服务配置
# ==========================================
# AI 服务提供商（zhipu, openai 等）
AI_PROVIDER=zhipu

# AI API Key（请替换为你的实际 API Key）
AI_API_KEY=your_ai_api_key_here

# AI API Base URL（可选，某些服务商需要）
AI_BASE_URL=

# ==========================================
# 邮件服务配置（用于问题反馈功能）
# ==========================================
# SMTP 服务器地址（QQ邮箱使用 smtp.qq.com）
SMTP_HOST=smtp.qq.com

# SMTP 端口（QQ邮箱使用 587）
SMTP_PORT=587

# 发送邮件的邮箱地址
SMTP_USER=your_email@qq.com

# 邮箱授权码（不是密码！）
# 获取方式：
# 1. 登录 QQ邮箱
# 2. 设置 -> 账户
# 3. 开启 SMTP 服务
# 4. 生成授权码并复制到这里
SMTP_PASS=your_email_auth_code

# ==========================================
# 部署说明
# ==========================================
# 1. 生产环境部署时，请务必修改以下配置：
#    - JWT_SECRET: 使用强随机字符串
#    - ADMIN_PASSWORD: 修改为强密码
#    - CORS_ORIGIN: 设置为实际的前端域名
#    - VITE_API_BASE_URL: 设置为实际的 API 地址
#    - VITE_SOCKET_URL: 设置为实际的 Socket.io 地址
#
# 2. 如果使用 Docker 部署，这些环境变量会在 docker-compose.yml 中配置
#
# 3. 如果使用传统方式部署，请在服务器上创建 .env 文件并配置这些变量
